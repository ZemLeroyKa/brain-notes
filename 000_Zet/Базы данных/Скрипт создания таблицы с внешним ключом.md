```sql
CREATE TABLE [dbo].[Groups] (
    Id INT IDENTITY(1,1) NOT NULL,
    Name NVARCHAR(50),
	FacultyId INT NOT NULL,
    CONSTRAINT PK_GroupID
        PRIMARY KEY NONCLUSTERED (Id),
    CONSTRAINT FK_Groups_Faculties FOREIGN KEY (FacultyId)
        REFERENCES Faculties(Id)
    ON DELETE CASCADE
    ON UPDATE CASCADE
);
```
